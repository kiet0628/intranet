#!/bin/bash

sudo -u ec2-user -i <<'EOF'
echo ".....From User Data" > $HOME/userdata.log

sudo yum install git -y

sudo mkdir -p /data; sudo chown ec2-user /data; 
cd /data; 

git clone https://github.com/kiet0628/intranet.git >> userdata.log

cd intranet/scripts
./installAll.sh >> $HOME/userdata.log

echo ".....End User Data" >> $HOME/userdata.log

EOF
